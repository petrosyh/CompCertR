Lemma asm_id
      (asm: Asm.program)
      (WF: Sk.wf asm.(module)):
    exists mp,
      (<<SIM: @ModPair.sim SimMemId.SimMemId SimMemId.SimSymbId SoundTop.Top mp>>)
